ra_fild12,295,208,0	script	飞空艇入口#rachel	45,1,1,{
	end;
OnTouch:
	mes "如果要搭乘飞空艇的话, ";
	mes "需要 1,200 Zeny来购买飞艇票.";
	mes "请问是否愿意搭乘飞空艇呢?";
	next;
	if (select("是的:不要") == 1) {
		if (countitem(7311) > 0) {
			delitem 7311,1; //Free_Flying_Ship_Ticket
			warp "airplane_01",245,60;
			end;
		}
		if (Zeny >= 1200) {
			set Zeny, Zeny - 1200;
			warp "airplane_01",245,60;
			end;
		}
		mes "抱歉, 您身上的 Zeny 不够.";
		mes "请记住我们的费用是 1,200 Zeny.";
		close;
	}
	mes "谢谢您选择了我们的服务. 欢迎下次再来.";
	close;
}
